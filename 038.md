#### Aşağıdaki C kodlarında kullanlan if deyimlerindeki hataları açıklayınız:

```c
double foo(void);
void bar(double);

int main(void)
{
	double dval = foo();

	if (dval > 3,4) // virgül operator
		bar(dval);
}
//////////////////////////////////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////////////////////////////////
double foo(void);
void bar(double);

int main(void)
{
	double dval = foo();

	if (dval > 4.5); //null statement
		bar(dval);
}
//////////////////////////////////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////////////////////////////////
int func(void);
void bar(int);

int main(void)
{
	int x = func();

	if (10 < x < 20) // her zaman doğru
		bar(x);
}
//////////////////////////////////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////////////////////////////////
int func(void);
void bar(int);

int main(void)
{
	int x = func();

	if (x != 7  || x != 23) // her zaman doğrı && olmalıydı.
		bar(x);
}

```
